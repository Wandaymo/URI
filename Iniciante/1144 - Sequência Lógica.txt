#include <stdio.h>
#include <stdlib.h>
#include <math.h>

int main()
{
    int n, i, x = 1, y = 1;

    scanf("%d",&n);

    for(i = 1; i <=n; i++)
    {
        x = pow(i,2);
        y = pow(i,3);

        printf("%d %d %d\n",i,x,y);

        x++;
        y++;

        printf("%d %d %d\n",i, x, y);

    }

    return 0;
}
