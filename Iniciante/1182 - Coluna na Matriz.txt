#include <stdio.h>
#include <stdlib.h>

int main()
{
    int N, i, j;
    char C;
    float M[12][12], soma = 0;


    scanf("%d ", &N);
    scanf("%c", &C);

    for(i = 0; i < 12; i++)
    {
        for(j = 0; j < 12; j++)
        {

            scanf("%f",&M[i][j]);
        }


    }
    if(C == 'S')
    {
        for(i = 0; i < 12; i++)
        {
            soma = soma + M[i][N];
        }
    }
    else if (C == 'M')
    {
        for(i = 0; i < 12; i++)
        {
            soma = soma + M[i][N];
        }
        soma = soma / 12;

    }
    printf("%.1f\n",soma);



    return 0;
}
